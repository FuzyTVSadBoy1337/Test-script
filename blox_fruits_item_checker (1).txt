-- Sword / Gun Checker
local function checkWeapons()
    local success, inventory = pcall(function()
        -- C√≥ th·ªÉ l√† "getInventory" thay v√¨ "getInventoryWeapons"
        return CommF:InvokeServer("getInventoryWeapons") 
    end)

    if not success or not inventory then
        warn("‚ùå Kh√¥ng l·∫•y ƒë∆∞·ª£c inventory weapons (c√≥ th·ªÉ remote ƒë·ªïi t√™n)")
        return
    end

    print("=== Weapons Found ===")
    for _, item in ipairs(inventory) do
        local name = item.Name or "Unknown"
        if item.Type == "Sword" then
            print("‚öîÔ∏è Sword:", name)
        elseif item.Type == "Gun" then
            print("üî´ Gun:", name)
        else
            print("‚ùì Unknown:", name, item.Type)
        end
    end
end

-- Melee Checker
local alias = {
    Combat           = {}, -- m·∫∑c ƒë·ªãnh c√≥
    DarkStep         = {"BuyBlackLeg", "BlackLeg"},
    Electro          = {"BuyElectro", "Electro"},
    WaterKungFu      = {"BuyFishmanKarate", "FishmanKarate"},
    DragonBreath     = {"BuyDragonBreath", "DragonBreath"},
    Superhuman       = {"BuySuperhuman", "Superhuman"},
    DeathStep        = {"BuyDeathStep", "DeathStep"},
    SharkmanKarate   = {"BuySharkmanKarate", "SharkmanKarate"},
    ElectricClaw     = {"BuyElectricClaw", "ElectricClaw"},
    DragonTalon      = {"BuyDragonTalon", "DragonTalon"},
    Godhuman         = {"BuyGodhuman", "Godhuman"},
}

local function checkMelee()
    print("=== Checking Melee ===")
    print("‚úÖ Owned Fighting Style: Combat (default)")

    for style, remotes in pairs(alias) do
        if style ~= "Combat" then
            local owned = false
            for _, remoteName in ipairs(remotes) do
                local ok, result = pcall(function()
                    return CommF:InvokeServer(remoteName, true)
                end)
                if ok and result == 1 then
                    owned = true
                    break
                end
            end

            if owned then
                print("‚úÖ Owned Fighting Style:", style)
            else
                print("‚ùå Not owned:", style)
            end
        end
    end
end

-- RUN
checkWeapons()
checkMelee()

print("=== Script End ===")
